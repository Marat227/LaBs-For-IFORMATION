//Сортировка Хоара
void Hoar(int arr[], int first, int last)
{
    int mid, count;
    int f = first, l = last;
    mid = arr[(f + l) / 2];
    do
    {
        while (arr[f] < mid) f++;
        while (arr[l] > mid) l--;
        if (f <= l)
        {
            count = arr[f];
            arr[f] = arr[l];
            arr[l] = count;
            f++;
            l--;
        }
    } while (f < l);
    if (first < l) Hoar(arr, first, l);
    if (f < last) Hoar(arr, f, last);
}

//Сортировка Шелла
void Shell(int arr[], int n) {
    int d = n;
    d = d / 2;
    int temp = 0;
    while (d > 0) {
        for (int i = 0; i < n - d; i++) {
            int j = i;

            while (j >= 0 && arr[j] > arr[j + d]) {
                temp = arr[j];
                arr[j] = arr[j + d];
                arr[j + d] = temp;
                j--;
            }
        }
        d = d / 2;
    }
    for (int k = 0; k < n; k++) cout << arr[k] <<' '; //вывод массива 

}
